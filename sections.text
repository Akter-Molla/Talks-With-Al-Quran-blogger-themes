// Fallback method using direct JSON fetch
async function islamicFetchBlogPostsFallback() {
    try {
        console.log('ফলব্যাক মেথড ব্যবহার করা হচ্ছে...');
        
        // Try multiple possible JSON feed URLs
        const feedUrls = [
            'https://chatwithalquran.blogspot.com/feeds/posts/default?alt=json&max-results=50',
            'https://chatwithalquran.blogspot.com/feeds/posts/default?alt=json-in-script&max-results=50',
            'https://chatwithalquran.blogspot.com/feeds/posts/default?alt=json&callback=?&max-results=50'
        ];
        
        let success = false;
        
        for (const feedUrl of feedUrls) {
            try {
                console.log(`চেষ্টা করছি: ${feedUrl}`);
                const response = await fetch(feedUrl);
                
                if (response.ok) {
                    const data = await response.json();
                    islamicBlogPosts = islamicParseBlogData(data);
                    islamicBlogLoaded = true;
                    success = true;
                    console.log(`সফলভাবে লোড হয়েছে ${islamicBlogPosts.length}টি পোস্ট`);
                    break;
                }
            } catch (error) {
                console.log(`URL এ ত্রুটি: ${feedUrl}`, error);
                continue;
            }
        }
        
        if (!success) {
            throw new Error('সকল URL ব্যর্থ হয়েছে');
        }
        
        // Update stats
        const statsElement = document.getElementById('islamic-blog-stats');
        if (statsElement) {
            statsElement.textContent = `${islamicBlogPosts.length}টি পোস্ট লোড হয়েছে`;
        }
        
    } catch (error) {
        console.error('ফলব্যাক মেথডেও ত্রুটি:', error);
        islamicLoadSampleBlogPosts();
    }
}

// Parse JSON feed data
function islamicParseBlogData(data) {
    const posts = [];
    
    try {
        console.log('ডাটা পার্স করা হচ্ছে...', data);
        
        if (data.feed && data.feed.entry) {
            data.feed.entry.forEach(entry => {
                try {
                    const title = entry.title?.$t || 'শিরোনাম নেই';
                    let content = '';
                    
                    // Extract content from different possible fields
                    if (entry.content) {
                        content = entry.content.$t || '';
                    } else if (entry.summary) {
                        content = entry.summary.$t || '';
                    }
                    
                    // Clean HTML tags from content for preview
                    const cleanContent = content.replace(/<[^>]*>/g, '').substring(0, 200);
                    
                    // Get post link
                    let link = '';
                    if (entry.link) {
                        const alternateLink = entry.link.find(link => link.rel === 'alternate');
                        if (alternateLink) {
                            link = alternateLink.href;
                        }
                    }
                    
                    // Get published date
                    const published = entry.published?.$t ? 
                        new Date(entry.published.$t).toLocaleDateString('bn-BD') : 
                        'তারিখ নেই';
                    
                    // Extract tags/categories
                    const tags = [];
                    if (entry.category) {
                        entry.category.forEach(cat => {
                            if (cat.term) {
                                tags.push(cat.term);
                            }
                        });
                    }
                    
                    // Extract first image if exists
                    let imageUrl = '';
                    if (content.includes('<img')) {
                        const imgMatch = content.match(/<img[^>]+src="([^">]+)"/);
                        if (imgMatch) {
                            imageUrl = imgMatch[1];
                        }
                    }
                    
                    if (title && content) {
                        posts.push({
                            title: title,
                            content: cleanContent + (cleanContent.length >= 200 ? '...' : ''),
                            fullContent: content,
                            link: link,
                            published: published,
                            tags: tags,
                            image: imageUrl
                        });
                    }
                } catch (entryError) {
                    console.error('এন্ট্রি পার্স করতে ত্রুটি:', entryError);
                }
            });
        }
        
        console.log(`পার্স করা হয়েছে ${posts.length}টি পোস্ট`);
        
    } catch (parseError) {
        console.error('ডাটা পার্স করতে ত্রুটি:', parseError);
    }
    
    // If no posts found, use sample data
    if (posts.length === 0) {
        console.log('কোন পোস্ট পাওয়া যায়নি, স্যাম্পল ডাটা ব্যবহার করা হচ্ছে');
        return islamicGetSampleBlogPosts();
    }
    
    return posts;
}

function islamicSearchInBlogPosts(searchTerm) {
    return islamicBlogPosts.filter(post => {
        const searchableText = (
            post.title.toLowerCase() + ' ' + 
            post.content.toLowerCase() + ' ' + 
            (post.fullContent || '').toLowerCase() + ' ' + 
            post.tags.join(' ').toLowerCase()
        );
        
        // Split search term into words for better matching
        const searchWords = searchTerm.split(/\s+/);
        return searchWords.some(word => searchableText.includes(word));
    });
}

    results.forEach((post, index) => {
        const highlightedContent = islamicHighlightText(post.content, searchTerm);
        const highlightedTitle = islamicHighlightText(post.title, searchTerm);
        
        resultsHTML += `
            <div class="islamic-blog-post" onclick="islamicViewPost(${islamicBlogPosts.indexOf(post)})" style="cursor: pointer;">

                <div style="margin-bottom: 10px; color: rgba(255,255,255,0.9);">
                    ${highlightedContent}
                </div>
                
                <div style="display: flex; justify-content: space-between; align-items: center; margin-top: 15px;">
                    <div style="display: flex; flex-wrap: wrap; gap: 5px;">
                        <small style="color: #f39c12;">📅 ${post.published}</small>
                        ${post.tags.slice(0, 3).map(tag => 
                            `<span class="islamic-tag-small" onclick="event.stopPropagation(); islamicSearchByTag('${tag}')">${tag}</span>`
                        ).join('')}
                    </div>
                    <span style="background: #27ae60; color: white; padding: 5px 12px; border-radius: 15px; font-size: 0.8rem; cursor: pointer;">
                        পড়ুন →
                    </span>
                </div>
            </div>
        `;
    });
    
    resultsHTML += `</div>`;
    
    // Add related searches
    resultsHTML += `
        <div style="margin-top: 40px; padding: 20px; background: rgba(255,255,255,0.05); border-radius: 10px;">
            <h5 style="color: #f1c40f; margin-bottom: 15px;">সম্পর্কিত সার্চ:</h5>
            <div style="display: flex; flex-wrap: wrap; gap: 10px;">
                <span class="islamic-tag" onclick="islamicSearchByTag('ইসলাম')">ইসলাম</span>
                <span class="islamic-tag" onclick="islamicSearchByTag('ইবাদত')">ইবাদত</span>
                <span class="islamic-tag" onclick="islamicSearchByTag('ফজিলত')">ফজিলত</span>
                <span class="islamic-tag" onclick="islamicSearchByTag('সওয়াব')">সওয়াব</span>
                <span class="islamic-tag" onclick="islamicSearchByTag('দোয়া')">দোয়া</span>
            </div>
        </div>
    `;
    
    blogContent.innerHTML = resultsHTML;
}

function islamicHighlightText(text, searchTerm) {
    if (!searchTerm || !text) return text;
    
    const searchWords = searchTerm.split(/\s+/);
    let highlightedText = text;
    
    searchWords.forEach(word => {
        if (word.length > 2) { // Only highlight words longer than 2 characters
            const regex = new RegExp(`(${word})`, 'gi');
            highlightedText = highlightedText.replace(regex, '<mark>$1</mark>');
        }
    });
    
    return highlightedText;
}

// Filter suras based on search
function islamicFilterSuras() {
    const searchTerm = islamicSurasSearch.value.toLowerCase();
    
    if (searchTerm === '') {
        islamicLoadSuraList();
        return;
    }
    
    const filteredSuras = islamicSuraNames.filter(sura => 
        sura.name.toLowerCase().includes(searchTerm)
    );
    
    let suraListHTML = '';
    filteredSuras.forEach(sura => {
        suraListHTML += `
            <div class="islamic-sura-item" onclick="islamicOpenSura(${sura.id})">

// Quran Data Functions
async function islamicFetchData() {
    try {
        islamicShowTypingIndicator();
        
        const [quranRes, arabicRes, tafsirRes] = await Promise.all([
            fetch("https://raw.githubusercontent.com/Akter-Molla/Quran-bot-bn/main/ayats_bn.json"),
            fetch("https://raw.githubusercontent.com/alQuranBD/Database/master/json/ayats_ar.json"),
            fetch("https://raw.githubusercontent.com/Akter-Molla/Quran-bot-bn/main/Tafseer_ibn_Kathir.json")
        ]);
        
        islamicQuranData = await quranRes.json();
        islamicArabicData = await arabicRes.json();
        islamicTafsirData = await tafsirRes.json();
        
        islamicHideTypingIndicator();
        islamicAddBotMessage('✅ কুরআন ডাটা সফলভাবে লোড হয়েছে! এখন আপনি কুরআনের আয়াত সম্পর্কে প্রশ্ন করতে পারেন।');
        
    } catch (error) {
        console.error("ডাটা লোড করতে ত্রুটি:", error);
        islamicAddBotMessage('❌ ডাটা লোড করতে সমস্যা হয়েছে। দয়া করে পরে আবার চেষ্টা করুন।');
    }
}

function islamicConvertNumbers(input) {
    const bnToEn = {'০':'0','১':'1','২':'2','৩':'3','৪':'4','৫':'5','৬':'6','৭':'7','৮':'8','৯':'9'};
    let output = input;
    for (const [bn, en] of Object.entries(bnToEn)) {
        output = output.replace(new RegExp(bn, 'g'), en);
    }
    return output;
}

function islamicConvertToBengaliNumerals(number) {
    const numerals = {'0':'০','1':'১','2':'২','3':'৩','4':'৪','5':'৫','6':'৬','7':'৭','8':'৮','9':'৯'};
    return number.toString().split('').map(digit => numerals[digit] || digit).join('');
}

function islamicSearchQuran(query) {
    const normalizedQuery = islamicConvertNumbers(query);
    const suraAyaMatch = normalizedQuery.match(/^(\d+)[:\.\/,](\d+)$/);
    
    if (suraAyaMatch) {
        const [, sura, aya] = suraAyaMatch;
        const exactMatch = islamicQuranData.find(a => a.sura === sura && a.aya === aya);
        return exactMatch ? [exactMatch] : [];
    }
    
    const words = normalizedQuery.toLowerCase().split(/\s+/);
    return islamicQuranData.filter(aya => {
        const text = aya.text.toLowerCase();
        return words.some(word => text.includes(word));
    });
}

function islamicShowQuizQuestion() {
    const questionData = islamicQuizData.currentQuiz[islamicQuizData.currentQuestion];

    let optionsHTML = '';

            </div>
        `;
    });
    
    // Start timer for this question
    islamicStartQuizTimer();
    
    questionsDiv.innerHTML = `

        </div>
        <div class="islamic-quiz-options">
            ${optionsHTML}
        </div>
    `;
    
    document.getElementById('islamic-quiz-score').textContent = 
        `স্কোর: ${islamicQuizData.score}/${islamicQuizData.totalQuestions}`;
}

function islamicStartQuizTimer() {
    islamicQuizData.timeLeft = 30; // 30 seconds per question
    
    islamicQuizData.timer = setInterval(() => {
        islamicQuizData.timeLeft--;
        
        const timerElement = document.getElementById('islamic-quiz-timer');
        if (timerElement) {
            timerElement.innerHTML = islamicGetTimerText(islamicQuizData.timeLeft);
        }
        
        if (islamicQuizData.timeLeft <= 0) {
            clearInterval(islamicQuizData.timer);
            islamicCheckAnswer(-1); // Time's up
        }
    }, 1000);
}

function islamicGetTimerText(seconds) {
    const timerFormat = document.getElementById('islamic-quiz-timer-format')?.value || 'bangla';
    
    if (timerFormat === 'bangla') {
        return `সময়: ${islamicConvertToBengaliNumerals(seconds)} সেকেন্ড`;
    } else {
        return `Time: ${seconds} seconds`;
    }
}

function islamicCheckAnswer(selectedIndex) {
    clearInterval(islamicQuizData.timer);
    
    const questionData = islamicQuizData.currentQuiz[islamicQuizData.currentQuestion];
    const options = document.querySelectorAll('.islamic-quiz-option');
    
    options.forEach((option, index) => {
        option.style.pointerEvents = 'none';
        if (index === questionData.correct) {
            option.classList.add('correct');
        } else if (index === selectedIndex && index !== questionData.correct) {
            option.classList.add('incorrect');
        }
    });
    
    if (selectedIndex === questionData.correct) {
        islamicQuizData.score++;
    }
    
    setTimeout(() => {
        islamicQuizData.currentQuestion++;
        if (islamicQuizData.currentQuestion < islamicQuizData.totalQuestions) {
            islamicShowQuizQuestion();
        } else {
            islamicShowQuizResults();
        }
    }, 2000);
}

function islamicShowQuizResults() {
    const questionsDiv = document.getElementById('islamic-quiz-questions');
    const percentage = (islamicQuizData.score / islamicQuizData.totalQuestions) * 100;
    
    let message = '';
    if (percentage >= 80) {
        message = 'অভিনন্দন! আপনি অসাধারণ করেছেন! 🎉';
    } else if (percentage >= 60) {
        message = 'ভালো করেছেন! 👍';
    } else {
        message = 'আরও চেষ্টা করুন, ইনশাআল্লাহ ভালো করবেন! 💪';
    }
    
    questionsDiv.innerHTML = `
        <div style="text-align: center; padding: 40px 20px;">
            <h3>কুইজ সম্পন্ন!</h3>
            <div style="font-size: 3rem; margin: 20px 0;">${islamicQuizData.score}/${islamicQuizData.totalQuestions}</div>
            <div style="font-size: 1.2rem; color: #f1c40f; margin-bottom: 20px;">${message}</div>
            <button class="islamic-quiz-category-btn" onclick="islamicResetQuiz()">আবার চেষ্টা করুন</button>
        </div>
    `;
}

// Voice Search
function islamicToggleVoiceSearch() {
    if (!islamicVoiceRecognition) {
        islamicInitializeVoiceRecognition();
    }
    
    if (islamicVoiceRecognition.isListening) {
        islamicStopVoiceRecognition();
    } else {
        islamicStartVoiceRecognition();
    }
}

function islamicInitializeVoiceRecognition() {
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    
    if (!SpeechRecognition) {
        islamicAddBotMessage('❌ আপনার ব্রাউজার ভয়েস রিকগনিশন সাপোর্ট করে না');
        return;
    }
    
    islamicVoiceRecognition = new SpeechRecognition();
    islamicVoiceRecognition.continuous = false;
    islamicVoiceRecognition.interimResults = false;
    islamicVoiceRecognition.lang = 'bn-BD';
    islamicVoiceRecognition.isListening = false;
    
    islamicVoiceRecognition.onstart = function() {
        islamicVoiceRecognition.isListening = true;
        islamicVoiceStatus.textContent = '🎤 শুনছি... কথা বলুন';
        islamicVoiceToggle.style.background = 'rgba(231, 76, 60, 0.8)';
    };
    
    islamicVoiceRecognition.onresult = function(event) {
        const transcript = event.results[0][0].transcript;
        if (islamicIsFullscreen) {
            document.getElementById('islamic-fullscreen-input').value = transcript;
        } else {
            document.getElementById('islamic-chat-input').value = transcript;
        }
        islamicVoiceStatus.textContent = `শোনা হয়েছে: "${transcript}"`;
    };
    
    islamicVoiceRecognition.onerror = function(event) {
        islamicVoiceStatus.textContent = 'ত্রুটি: আবার চেষ্টা করুন';
        islamicVoiceRecognition.isListening = false;
        islamicVoiceToggle.style.background = '';
    };
    
    islamicVoiceRecognition.onend = function() {
        islamicVoiceRecognition.isListening = false;
        islamicVoiceToggle.style.background = '';
        setTimeout(() => {
            islamicVoiceStatus.textContent = '';
        }, 3000);
    };
}

function islamicStartVoiceRecognition() {
    try {
        islamicVoiceRecognition.start();
    } catch (error) {
        islamicVoiceStatus.textContent = 'ভয়েস শুরু করতে সমস্যা';
    }
}

function islamicStopVoiceRecognition() {

  function validateForm() {
    const name = document.getElementById('participantName').value.trim();
    const email = document.getElementById('participantEmail').value.trim();
    const phone = document.getElementById('participantPhone').value.trim();
    let isValid = true;
    
    if (name === '') {
      document.getElementById('nameError').textContent = 'নাম অবশ্যই প্রয়োজন';
      isValid = false;
    } else if (name.length < 2) {
      document.getElementById('nameError').textContent = 'নাম খুব ছোট';
      isValid = false;
    } else {
      document.getElementById('nameError').textContent = '';
    }
    
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (email === '') {
      document.getElementById('emailError').textContent = 'ইমেইল অবশ্যই প্রয়োজন';
      isValid = false;
    } else if (!emailRegex.test(email)) {
      document.getElementById('emailError').textContent = 'সঠিক ইমেইল দিন';
      isValid = false;
    } else {
      document.getElementById('emailError').textContent = '';
    }
    
    const phoneRegex = /^(\+?88)?01[3-9]\d{8}$/;
    if (phone !== '' && !phoneRegex.test(phone)) {
      document.getElementById('phoneError').textContent = 'সঠিক ফোন নম্বর দিন';
      isValid = false;
    } else {
      document.getElementById('phoneError').textContent = '';
    }
    
    // Get selected difficulty
    const difficultyOptions = document.querySelectorAll('input[name="difficulty"]');
    for (const option of difficultyOptions) {
      if (option.checked) {
        difficulty = option.value;
        break;
      }
    }
    
    return isValid;
  }

  function startQuiz() {
    if (!validateForm()) return;
    
    participantName = document.getElementById('participantName').value.trim();
    participantEmail = document.getElementById('participantEmail').value.trim();
    participantPhone = document.getElementById('participantPhone').value.trim();
    
    // Reset quiz variables
    currentQuestion = 0;
    score = 0;
    correctCount = 0;
    wrongCount = 0;
    skippedCount = 0;
    userAnswers = [];
    hintsUsed = 0;
    isPaused = false;
    
    document.getElementById('registrationSection').style.display = 'none';
    document.getElementById('quizContainer').style.display = 'flex';
    document.getElementById('quizRightColumn').style.display = 'flex';
    
    document.getElementById('welcomeMessage').textContent = `স্বাগতম, ${participantName}!`;
    document.getElementById('highScore').textContent = highScore;
    
    // Filter questions based on difficulty
    let filteredQuestions = [...questions];
    if (difficulty === 'easy') {
      // Take only questions from first 3 categories
      filteredQuestions = questions.filter(q => 
        ['প্রাথমিক জীবন', 'নবুওয়াত', 'হিজরত'].includes(q.category)
      );
    } else if (difficulty === 'medium') {
      // Take questions from first 4 categories
      filteredQuestions = questions.filter(q => 
        ['প্রাথমিক জীবন', 'নবুওয়াত', 'হিজরত', 'যুদ্ধ'].includes(q.category)
      );
    }
    // For hard difficulty, use all questions
    
    // Shuffle and limit questions based on difficulty
    filteredQuestions = shuffleArray(filteredQuestions);
    if (difficulty === 'easy') {
      filteredQuestions = filteredQuestions.slice(0, 10);
    } else if (difficulty === 'medium') {
      filteredQuestions = filteredQuestions.slice(0, 15);
    } else {
      filteredQuestions = filteredQuestions.slice(0, 20);
    }
    
    questions = filteredQuestions;
    
    updateProgress();
    loadQuestion();
  }

  function shuffleArray(array) {
    for (let i = array.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [array[i], array[j]] = [array[j], array[i]];
    }
    return array;
  }

  function updateProgress() {
    const progress = (currentQuestion / questions.length) * 100;
    document.getElementById('progressFill').style.width = `${progress}%`;
    document.getElementById('progressText').textContent = `${Math.round(progress)}%`;
    document.getElementById('currentScore').textContent = score;
    document.getElementById('correctAnswers').textContent = correctCount;
    document.getElementById('wrongAnswers').textContent = wrongCount;
    document.getElementById('skippedAnswers').textContent = skippedCount;
  }

  function startTimer() {
    timeLeft = 30;
    const timerText = document.getElementById('timerText');
    const timerProgress = document.getElementById('timerProgress');
    
    const circumference = 2 * Math.PI * 15.9155;
    timerProgress.style.strokeDasharray = `${circumference} ${circumference}`;
    timerProgress.style.strokeDashoffset = circumference;
    
    timerInterval = setInterval(() => {
      if (isPaused) return;
      
      timeLeft--;
      timerText.textContent = timeLeft;
      
      const offset = circumference - (timeLeft / 30) * circumference;
      timerProgress.style.strokeDashoffset = offset;
      
      if (timeLeft <= 10) {
        timerProgress.style.stroke = '#ff6b6b';
      } else if (timeLeft <= 20) {
        timerProgress.style.stroke = '#ffa726';
      } else {
        timerProgress.style.stroke = '#00a870';
      }
      
      if (timeLeft <= 0) {
        clearInterval(timerInterval);
        handleTimeUp();
      }
    }, 1000);
  }

  function loadQuestion() {
    if (currentQuestion >= questions.length) {
      showResults();
      return;
    }
    
    quizActive = true;
    const q = questions[currentQuestion];
    
    document.getElementById('questionCategory').textContent = q.category;
    document.getElementById('questionNumber').textContent = `প্রশ্ন ${currentQuestion + 1}/${questions.length}`;
    document.getElementById('questionText').textContent = q.q;
    
    const optionsGrid = document.getElementById('optionsGrid');
    optionsGrid.innerHTML = '';
    
    const shuffledOptions = shuffleArray([...q.options]);
    
    shuffledOptions.forEach((option, idx) => {
      const btn = document.createElement('button');
      btn.className = 'option-btn';
      btn.textContent = option;
      btn.dataset.value = option;
      btn.onclick = () => selectAnswer(option, btn);
      optionsGrid.appendChild(btn);
    });
    
    document.getElementById('explanationBox').style.display = 'none';
    document.getElementById('timeBonus').style.display = 'none';
    
    updateProgress();
    startTimer();
  }

  function selectAnswer(selectedOption, btn) {
    if (!quizActive || isPaused) return;
    
    clearInterval(timerInterval);
    quizActive = false;
    
    const q = questions[currentQuestion];
    const correctOption = q.options[q.answer];
    
    // Record user answer
    userAnswers[currentQuestion] = {
      question: q.q,
      userAnswer: selectedOption,
      correctAnswer: correctOption,
      isCorrect: selectedOption === correctOption,
      isSkipped: false,
      explanation: q.explanation
    };
    
    // সব অপশন ডিজেবল করুন
    document.querySelectorAll('.option-btn').forEach(opt => {
      opt.disabled = true;
      if (opt.dataset.value === correctOption) {
        opt.classList.add('correct');
      }
    });
    
    let timeBonus = 0;
    let showBonusMessage = false;
    
    // Calculate time bonus only for correct answers
    if (selectedOption === correctOption) {
      if (timeLeft > 20) {
        timeBonus = 2;
        showBonusMessage = true;
      } else if (timeLeft > 10) {
        timeBonus = 1;
        showBonusMessage = true;
      }
      
      // Show bonus message only for correct answers
      if (showBonusMessage) {
        document.getElementById('timeBonus').style.display = 'block';
      }
      
      btn.classList.add('correct');
      score += 1 + timeBonus;
      correctCount++;
      
      // ভিজ্যুয়াল ফিডব্যাক
      btn.style.transform = 'scale(1.05)';
      setTimeout(() => {
        btn.style.transform = 'scale(1)';
      }, 300);
    } else {
      btn.classList.add('wrong');
      wrongCount++;
      // Ensure bonus message is hidden for wrong answers
      document.getElementById('timeBonus').style.display = 'none';
    }
    
    // অটোমেটিকভাবে পরবর্তী প্রশ্নে যান
    setTimeout(() => {
      currentQuestion++;
      loadQuestion();
    }, 2000);
    
    showExplanation();
  }

  function handleTimeUp() {
    if (isPaused) return;
    
    quizActive = false;
    document.querySelectorAll('.option-btn').forEach(btn => btn.disabled = true);
    wrongCount++;
    
    // Record as skipped
    userAnswers[currentQuestion] = {
      question: questions[currentQuestion].q,
      userAnswer: null,
      correctAnswer: questions[currentQuestion].options[questions[currentQuestion].answer],
      isCorrect: false,
      isSkipped: true,
      explanation: questions[currentQuestion].explanation
    };
    
    setTimeout(() => {
      currentQuestion++;
      loadQuestion();
    }, 2000);
    
    showExplanation();
  }

  }

  function showHint() {
    if (hintsUsed >= 2 || !quizActive || isPaused) return;
    
    hintsUsed++;
    const q = questions[currentQuestion];
    const correctOption = q.options[q.answer];
    
    // Remove one wrong option
    const wrongOptions = q.options.filter(opt => opt !== correctOption);
    const optionToRemove = wrongOptions[Math.floor(Math.random() * wrongOptions.length)];
    
    document.querySelectorAll('.option-btn').forEach(btn => {
      if (btn.dataset.value === optionToRemove) {
        btn.disabled = true;
        btn.style.opacity = '0.5';
        btn.style.textDecoration = 'line-through';
      }
    });
    
    // Disable hint button after using it twice
    if (hintsUsed >= 2) {
      document.getElementById('hintBtn').disabled = true;
      document.getElementById('hintBtn').style.opacity = '0.5';
    }
  }

  function addToLeaderboard(name, score, email) {
    const existingEntry = leaderboard.find(entry => entry.email === email);
    
    if (existingEntry) {
      if (score > existingEntry.score) {
        existingEntry.score = score;
        existingEntry.name = name;
        existingEntry.date = new Date().toLocaleDateString('bn-BD');
      }
    } else {
      leaderboard.push({
        name: name,
        score: score,
        email: email,
        date: new Date().toLocaleDateString('bn-BD')
      });
    }
    
    leaderboard.sort((a, b) => b.score - a.score);
    leaderboard = leaderboard.slice(0, 10);
    localStorage.setItem('seerahLeaderboard', JSON.stringify(leaderboard));
  }

  function showLeaderboard() {
    document.getElementById('quizContainer').style.display = 'none';
    document.getElementById('quizRightColumn').style.display = 'none';
    document.getElementById('leaderboardSection').style.display = 'flex';
    
    updateLeaderboard();
  }

  function updateLeaderboard() {
    const leaderboardList = document.getElementById('leaderboardList');
    leaderboardList.innerHTML = '';
    
    if (leaderboard.length === 0) {
      leaderboardList.innerHTML = '<p>এখনও কোন স্কোর নেই। প্রথম হন!</p>';
      return;
    }
    
    // Filter leaderboard based on current tab
    let filteredLeaderboard = [...leaderboard];
    if (currentLeaderboardTab === 'daily') {
      const today = new Date().toLocaleDateString('bn-BD');
      filteredLeaderboard = leaderboard.filter(entry => entry.date === today);
    } else if (currentLeaderboardTab === 'weekly') {
      const oneWeekAgo = new Date();
      oneWeekAgo.setDate(oneWeekAgo.getDate() - 7);
      filteredLeaderboard = leaderboard.filter(entry => {
        const entryDate = new Date(entry.date);
        return entryDate >= oneWeekAgo;
      });
    }
    // For alltime, use all entries
    
    filteredLeaderboard.forEach((entry, index) => {
      const item = document.createElement('div');
      item.className = 'leaderboard-item';
      item.innerHTML = `
        <div class="leaderboard-rank">${index + 1}</div>
        <div class="leaderboard-name">${entry.name}</div>
        <div class="leaderboard-score">${entry.score}</div>
      `;
      leaderboardList.appendChild(item);
    });
  }

  function showReview() {
    document.getElementById('resultContainer').style.display = 'none';
    document.getElementById('qaReviewSection').style.display = 'flex';
    
    displayQAReview('all');
  }

  function displayQAReview(filter) {
    const qaList = document.getElementById('qaList');
    qaList.innerHTML = '';
    
    let filteredAnswers = [...userAnswers];
    
    if (filter === 'correct') {
      filteredAnswers = userAnswers.filter(answer => answer.isCorrect);
    } else if (filter === 'wrong') {
      filteredAnswers = userAnswers.filter(answer => !answer.isCorrect && !answer.isSkipped);
    } else if (filter === 'skipped') {
      filteredAnswers = userAnswers.filter(answer => answer.isSkipped);
    }
    
    if (filteredAnswers.length === 0) {
      qaList.innerHTML = '<p>এই ফিল্টারে কোন প্রশ্ন নেই।</p>';
      return;
    }
    
    filteredAnswers.forEach((answer, index) => {
      const originalIndex = userAnswers.indexOf(answer);
      const item = document.createElement('div');
      item.className = 'qa-item';
      
      let statusClass = '';
      let statusText = '';
      
      if (answer.isCorrect) {
        statusClass = 'correct';
        statusText = 'সঠিক';
      } else if (answer.isSkipped) {
        statusClass = 'skipped';
        statusText = 'স্কিপ';
      } else {
        statusClass = 'wrong';
        statusText = 'ভুল';
      }
      
      item.innerHTML = `
        <div class="qa-item-header">
          <div class="qa-item-number">প্রশ্ন ${originalIndex + 1}</div>
          <div class="qa-item-status ${statusClass}">${statusText}</div>
        </div>
        <div class="qa-item-question">${answer.question}</div>
        <div class="qa-item-answers">
          ${answer.userAnswer ? `<div class="qa-item-answer qa-item-user-answer">আপনার উত্তর: ${answer.userAnswer}</div>` : ''}
          <div class="qa-item-answer qa-item-correct-answer">সঠিক উত্তর: ${answer.correctAnswer}</div>
        </div>
        <div class="qa-item-explanation">${answer.explanation}</div>
      `;
      
      qaList.appendChild(item);
    });
  }

  function filterQAReview(filter) {
    displayQAReview(filter);
  }

  function shareResults() {
    const shareText = `আমি সিরাত কুইজে ${score}/${questions.length} স্কোর করেছি! ${participantName} হিসেবে অংশগ্রহণ করেছি।`;
    
    if (navigator.share) {
      navigator.share({
        title: 'সিরাত কুইজ ফলাফল',
        text: shareText,
        url: window.location.href
      });
